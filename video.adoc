<script>
// ═══════════════════════════════════════════════════════
// ВИДЕОМАТЕРИАЛЫ — заполните массив:
//   [номер, "Название видео", "ссылка"]
// ═══════════════════════════════════════════════════════
const VIDEOS = [
  [1, "Маршрут на г.Сельбухра, маршрут по промоине Салачик", "https://rutube.ru/video/607ee0629e36f47cf219cddf42851b74/"],
  [2, "Маршрут на г.Беш-Кош", "https://rutube.ru/video/16c2c63417c7c990e45d49a54bb3b0b1/"],
  [3, "Маршрут на г.Сувлу-кая", "https://rutube.ru/video/9b0ff13997708b9214de8212fdebd9ec/"]
   [4, "Маршрут на Верхоречье", "https://rutube.ru/video/795d054416695e4bb1a86f21a04a52d6/"]
  // ← добавляйте новые строки
];
// ═══════════════════════════════════════════════════════

function taRender() {
  const q = (document.getElementById('ta-q').value||'').toLowerCase();
  const vis = VIDEOS.filter(([n, title]) => {
    if (q && !title.toLowerCase().includes(q) && !String(n).includes(q)) return false;
    return true;
  });
  const tot = VIDEOS.length;
  document.getElementById('ta-stats').innerHTML =
    `Всего видео: <b>${tot}</b>` + (q ? ` · Найдено: <b>${vis.length}</b>` : '');
  const out = document.getElementById('ta-out');
  if (!vis.length) {
    out.innerHTML = '<div class="ta-wrap"><div class="ta-empty">Ничего не найдено.</div></div>';
    return;
  }
  out.innerHTML = `<div class="ta-wrap"><table class="ta-tbl">
    <thead><tr><th>№</th><th>Название</th><th>Ссылка</th></tr></thead>
    <tbody>${vis.map(([n, title, url])=>`
      <tr>
        <td class="td-n">${n}</td>
        <td>${title}</td>
        <td class="td-l"><a class="ta-link" href="${url}" target="_blank" rel="noopener">Смотреть</a></td>
      </tr>`).join('')}
    </tbody></table></div>`;
}

taRender();
</script>
