let flt = 'all';

function taFilter(f) {
  flt = f;
  ['all','free','taken'].forEach(id =>
    document.getElementById('tf-'+id).classList.toggle('on', id===f));
  taRender();
}

function taRender() {
  const q = (document.getElementById('ta-q').value||'').toLowerCase();
  const vis = TOPICS.filter(([n, taken, title]) => {
    if (flt==='free'  && taken)  return false;
    if (flt==='taken' && !taken) return false;
    if (q && !title.toLowerCase().includes(q) && !String(n).includes(q)) return false;
    return true;
  });
  const tot  = TOPICS.length;
  const free = TOPICS.filter(([,t])=>!t).length;
  document.getElementById('ta-stats').innerHTML =
    `Всего: <b>${tot}</b> &nbsp;·&nbsp; Свободно: <b>${free}</b> &nbsp;·&nbsp; Занято: <b>${tot-free}</b>`;
  const out = document.getElementById('ta-out');
  if (!vis.length) {
    out.innerHTML = '<div class="ta-wrap"><div class="ta-empty">Темы не найдены.</div></div>';
    return;
  }
  out.innerHTML = `<div class="ta-wrap"><table class="ta-tbl">
    <thead><tr><th>№</th><th>Название темы</th><th>Статус</th></tr></thead>
    <tbody>${vis.map(([n, taken, title])=>`
      <tr class="${taken?'taken':''}">
        <td class="td-n">${n}</td>
        <td>${title}</td>
        <td class="td-s"><span class="badge ${taken?'badge-t':'badge-f'}">${taken?'Занято':'Свободно'}</span></td>
      </tr>`).join('')}
    </tbody></table></div>`;
}

taRender();
</script>
