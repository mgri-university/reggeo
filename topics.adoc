let topics    = [];   // [{num, title, taken}]
let filter    = 'all';
let selected  = null; // {num, title}

async function init() {
  // Попытка загрузить статусы из Яндекс Таблицы
  if (SHEET_CSV_URL && SHEET_CSV_URL !== 'ВСТАВЬТЕ_СЮДА_ССЫЛКУ_НА_CSV') {
    try {
      const resp = await fetch(SHEET_CSV_URL);
      const csv  = await resp.text();
      topics = parseCSV(csv);
    } catch(e) {
      topics = STATIC_TOPICS.map(([num, title]) => ({ num, title, taken: false }));
    }
  } else {
    // Демо-режим: все темы свободны
    topics = STATIC_TOPICS.map(([num, title]) => ({ num, title, taken: false }));
  }
  renderTable();
}

function parseCSV(csv) {
  const rows = csv.trim().split('\n').slice(1); // пропускаем заголовок
  return rows.map(row => {
    // Ожидаемые столбцы: №, Название, Статус (Свободно/Занято)
    const cols = row.split(',');
    const num   = parseInt(cols[0]) || 0;
    const title = (cols[1] || '').replace(/^"|"$/g, '').trim();
    const stat  = (cols[2] || '').replace(/^"|"$/g, '').trim().toLowerCase();
    return { num, title, taken: stat === 'занято' };
  }).filter(t => t.num > 0);
}

function setFilter(f) {
  filter = f;
  ['all','free','taken'].forEach(id => {
    document.getElementById('btn-' + id).classList.toggle('active', id === f);
  });
  renderTable();
}

function renderTable() {
  const q = (document.getElementById('ta-search').value || '').toLowerCase();
  const filtered = topics.filter(t => {
    if (filter === 'free'  && t.taken)  return false;
    if (filter === 'taken' && !t.taken) return false;
    if (q && !t.title.toLowerCase().includes(q) && !String(t.num).includes(q)) return false;
    return true;
  });

  const total = topics.length;
  const free  = topics.filter(t => !t.taken).length;
  document.getElementById('ta-stats').innerHTML =
    `<span>Всего тем: <b>${total}</b></span>` +
    `<span>Свободно: <b>${free}</b></span>` +
    `<span>Занято: <b>${total - free}</b></span>`;

  const wrap = document.getElementById('ta-content');
  if (filtered.length === 0) {
    wrap.innerHTML = '<div class="ta-table-wrap"><div class="ta-empty">Темы не найдены. Попробуйте изменить фильтр или поисковый запрос.</div></div>';
    return;
  }

  const rows = filtered.map(t => `
    <tr class="${t.taken ? 'taken-row' : ''}">
      <td class="td-num">${t.num}</td>
      <td class="td-title">${t.title}</td>
      <td class="td-status"><span class="badge ${t.taken ? 'badge-taken' : 'badge-free'}">${t.taken ? 'Занято' : 'Свободно'}</span></td>
      <td class="td-action">
        <button class="btn-select" onclick="openModal(${t.num})" ${t.taken ? 'disabled title="Тема уже занята"' : ''}>
          Выбрать
        </button>
      </td>
    </tr>`).join('');

  wrap.innerHTML = `
    <div class="ta-table-wrap">
      <table class="ta-table">
        <thead>
          <tr>
            <th>№</th>
            <th>Название темы</th>
            <th>Статус</th>
            <th></th>
          </tr>
        </thead>
        <tbody>${rows}</tbody>
      </table>
    </div>`;
}

function openModal(num) {
  const t = topics.find(x => x.num === num);
  if (!t) return;
  selected = t;
  document.getElementById('modal-title').textContent = `Тема №${t.num}`;
  document.getElementById('modal-topic-preview').textContent = t.title;
  document.getElementById('modal-name').value = '';
  document.getElementById('btn-confirm').disabled = true;
  document.getElementById('ta-overlay').classList.add('open');
  setTimeout(() => document.getElementById('modal-name').focus(), 50);
}

function closeModal() {
  document.getElementById('ta-overlay').classList.remove('open');
  selected = null;
}

function validateModal() {
  const v = document.getElementById('modal-name').value.trim();
  document.getElementById('btn-confirm').disabled = v.length < 3;
}

function submitChoice() {
  if (!selected) return;
  const name = document.getElementById('modal-name').value.trim();
  if (name.length < 3) return;

  let url = FORM_URL;
  if (url && url !== 'ВСТАВЬТЕ_СЮДА_ССЫЛКУ_НА_ФОРМУ') {
    const params = new URLSearchParams();
    params.set(FORM_PARAM_NAME,  name);
    params.set(FORM_PARAM_TOPIC, `№${selected.num} — ${selected.title}`);
    url = url + (url.includes('?') ? '&' : '?') + params.toString();
    window.open(url, '_blank');
  } else {
    // Демо: просто показываем что было бы
    alert(`Демо-режим.\nФИО: ${name}\nТема: №${selected.num} — ${selected.title}\n\nВ реальном режиме здесь откроется Яндекс Форма.`);
  }
  closeModal();
}

// Закрыть по клику на фон
document.getElementById('ta-overlay').addEventListener('click', function(e) {
  if (e.target === this) closeModal();
});

// Закрыть по Escape
document.addEventListener('keydown', e => { if (e.key === 'Escape') closeModal(); });

init();
